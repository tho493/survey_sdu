<?php

use Illuminate\Http\Request;
use Illuminate\Support\Facades\Route;
use App\Http\Controllers\Api\KhaoSatApiController;

Route::prefix('v1')->group(function () {
    // Public API
    Route::get('khao-sat/active', [KhaoSatApiController::class, 'getActiveKhaoSat']);
    Route::get('khao-sat/{dotKhaoSat}', [KhaoSatApiController::class, 'show']);
    Route::post('khao-sat/{dotKhaoSat}/submit', [KhaoSatApiController::class, 'submit']);

    // Protected API
    Route::middleware('auth:api')->group(function () {
        Route::get('thong-ke/dot/{dotKhaoSat}', [KhaoSatApiController::class, 'thongKe']);
        Route::get('bao-cao/export/{dotKhaoSat}', [KhaoSatApiController::class, 'export']);
    });
});
